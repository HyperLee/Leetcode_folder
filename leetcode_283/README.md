public static void MoveZeroes(int[] nums)
{
    if (nums is null) throw new ArgumentNullException(nameof(nums));

    int pointer = 0;
    for (int i = 0; i < nums.Length; i++)
    {
        if (nums[i] != 0)
        {
            if (pointer != i)
            {
                nums[pointer] = nums[i];
                nums[i] = 0;
            }
            pointer++;
        }
    }
}
